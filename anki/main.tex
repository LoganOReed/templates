\begin{note}
  \xplain{837c36cc-8a5d-437a-b3b3-617d4a3a39e0}
  \tags{Definition}
  \xfield{A partition $\Gamma$ of an interval $[a,b] \in \mathbb{R}^{}$}
  \begin{field}
    a collection of points $\Gamma = \left\{ x_{0}, x_{1}, \ldots, x_{m} \right\}$ s.t. $x_{0} = a$, $x_{m} = b$, and $x_{i-1} < x_{i}$ for $i = 1,\ldots,m$.
  \end{field}
  \xfield{Notation for the set of all partitions of $[a,b]$}
  \begin{field}
    Denoted $\Omega_{a}^{b}$.
  \end{field}
\end{note}

\begin{note}
  \xplain{b36e1ac5-afca-4404-af5c-a186e98a567b}
  \tags{Definition}
  \xfield{Let $f : [a,b] \to \mathbb{R}^{} $. The variation of f over $[a,b]$ is defined as}
  \begin{field}
    $V = V[f;a,b] = \sup_{\Gamma \in \Omega_{a}^{b}}\left\{ \sum_{i=1}^{m} \left| f(x_{i}) - f(x_{i-1}) \right| \right\} $
  \end{field}
  \xfield{What is the notation for $\left| f(x_{i}) - f(x_{i-1}) \right|$?}
  \begin{field}
    $S_{\Gamma}([a,b])$.
  \end{field}
  \xfield{When is $f$ of bounded variation on $[a,b]$?}
  \begin{field}
    When $V < \infty$.
  \end{field}
\end{note}
